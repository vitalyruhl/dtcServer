cmake_minimum_required(VERSION 3.20)

# Simple test target for WebSocket connection
add_executable(test_websocket
    test_websocket_connection.cpp
)

# Link the required libraries
target_link_libraries(test_websocket
    coinbase_feed
    exchange_base
    dtc_util
    ws2_32  # Windows Socket library
)

# Include directories
target_include_directories(test_websocket PRIVATE
    ${CMAKE_SOURCE_DIR}/include
)
